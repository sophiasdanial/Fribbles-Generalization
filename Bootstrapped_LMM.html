##LMM for null model
library(lmerTest)

# Set the number of bootstrap simulations
n_simulations <- 1000

# Fit the original lmer model
null.0 <- lmerTest::lmer(Score.avg ~ (1|Subject), data = avgeff.df)

# Extract the residuals from the lmer model
null_res <- residuals(null.0)

# Create an empty vector to store the bootstrapped sample statistics
boot_null_samples <- vector(mode = "numeric", length = n_simulations)

# Set the seed for reproducibility
set.seed(123)

# Perform the wild bootstrap simulation
for (i in 1:n_simulations) {
  # Generate a vector of random integers to use as indices for the bootstrap
  null_random_indices <- sample.int(length(null_res), length(null_res), replace = TRUE)
  
  # Calculate the bootstrapped residuals (Rademacher)
  boot_null_resid <- null_res[null_random_indices] * sample(c(-1, 1), length(null_res), replace = TRUE)
  
  # Add the bootstrapped residuals to the fixed effects of the original model
  boot_null_score <- fitted(null.0) +boot_null_resid
  
  # Fit the bootstrapped model
  boot_null_model <- lmerTest::lmer(boot_null_score ~ (1|Subject), data = avgeff.df)
  
  # Extract the summary statistics of interest
  boot_null_samples[i] <- mean(fixef(boot_null_model))
}

lmm.0 <- lmerTest::lmer(Score.avg ~ Prime*SampleGroup*GS.avg + (1|Subject), data = avgeff.df)

# Extract the residuals from the lmer model
residuals <- residuals(lmm.0)

# Create an empty vector to store the bootstrapped sample statistics
boot_samples <- vector(mode = "numeric", length = n_simulations)

# Set the seed for reproducibility
set.seed(123)

# Perform the wild bootstrap simulation
for (i in 1:n_simulations) {
  # Generate a vector of random integers to use as indices for the bootstrap
  random_indices <- sample.int(length(residuals), length(residuals), replace = TRUE)
  
  # Calculate the bootstrapped residuals (Rademacher)
  boot_resid <- residuals[random_indices] * sample(c(-1, 1), length(residuals), replace = TRUE)
  
  # Add the bootstrapped residuals to the fixed effects of the original model
  boot_score <- fitted(lmm.0) + boot_resid
  
  # Fit the bootstrapped model
  boot_model <- lmerTest::lmer(boot_score ~ Prime*SampleGroup*GS.avg + (1|Subject), data = avgeff.df)
  
  # Extract the summary statistics of interest
  boot_samples[i] <- mean(fixef(boot_model))
}
